/*a Uma consulta usando junção de mais de 2 tabelas*/
select  z.nome, c.numero, c.bloco, c.andar from
tbpessoa z inner join tbpessoa_has_unidade b on
z.idpessoa = b.idtbpessoa inner join tbunidade c on
c.idunidade = b.idtbunidade;
/*b Uma consulta que seja útil para a lógica de negócios usando totalização e
uma função de data*/
/*Obter o nome e a quantidade de entradas que determinada pessoa fez ao condominio dentro das ultimas 24 horas*/


select  p.nome, count(a.idpessoa) as "Total de entradas" from
tbpessoa p inner join tbacesso a on
a.idpessoa = p.idpessoa
where a.datahorachegada between sysdate - 1 and sysdate
group by p.nome
order by "Total de entradas";

/*c. Uma consulta usando junção externa left join ou right join*/
/*Selecionar todos as pessoas para saber quais tem alguma relação com alguma unidade*/

tbpessoa z left outer join tbpessoa_has_unidade b on
z.idpessoa = b.idtbpessoa;
